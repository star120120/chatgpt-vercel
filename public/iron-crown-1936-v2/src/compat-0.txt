function clipHalf(poly,A,B,C){
 const out=[];if(!poly.length)return out;
 const inside=p=>A*p[0]+B*p[1]<=C+1e-7;
 const intersect=(p,q)=>{const dp=A*p[0]+B*p[1]-C,dq=A*q[0]+B*q[1]-C,t=dp/(dp-dq||1e-9);return[p[0]+(q[0]-p[0])*t,p[1]+(q[1]-p[1])*t]};
 for(let i=0;i<poly.length;i++){const p=poly[i],q=poly[(i+1)%poly.length],ip=inside(p),iq=inside(q);if(ip&&iq)out.push(q);else if(ip&&!iq)out.push(intersect(p,q));else if(!ip&&iq){out.push(intersect(p,q));out.push(q)}}return out;
}
function voronoiCell(i,pts){let poly=[[8,8],[W-8,8],[W-8,H-8],[8,H-8]],p=pts[i];for(let j=0;j<pts.length&&poly.length;j++){if(i===j)continue;const q=pts[j],A=2*(q[0]-p[0]),B=2*(q[1]-p[1]),C=q[0]*q[0]+q[1]*q[1]-p[0]*p[0]-p[1]*p[1];poly=clipHalf(poly,A,B,C)}return poly}
function cross(o,a,b){return(a[0]-o[0])*(b[1]-o[1])-(a[1]-o[1])*(b[0]-o[0])}
function convexHull(points){const p=[...points].sort((a,b)=>a[0]-b[0]||a[1]-b[1]);if(p.length<=2)return p;const lo=[],hi=[];for(const x of p){while(lo.length>=2&&cross(lo[lo.length-2],lo[lo.length-1],x)<=0)lo.pop();lo.push(x)}for(let i=p.length-1;i>=0;i--){const x=p[i];while(hi.length>=2&&cross(hi[hi.length-2],hi[hi.length-1],x)<=0)hi.pop();hi.push(x)}lo.pop();hi.pop();return lo.concat(hi)}
function expandedHull(points,margin){const expanded=[];const dirs=12;for(const p of points)for(let i=0;i<dirs;i++){const a=i*Math.PI*2/dirs;expanded.push([p[0]+Math.cos(a)*margin,p[1]+Math.sin(a)*margin])}return convexHull(expanded)}
